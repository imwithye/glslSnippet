// +iResolution: 100% 250px
// +iPlot: 0 0 10

void mainImage(out vec4 fragColor, in vec2 fragCoord)
{
    vec3 grid = vec3(0.3, 0.3, 0.3);
    plotg(fragCoord, grid);
    
    vec3 f1 = vec3(1., 1., 0.);
    plot(log, fragCoord, f1);
    
    vec3 f2 = vec3(1., 0., 0.);
    plot(sin, fragCoord, f2);
    
    vec3 f3 = vec3(0., 1., 1.);
    ploti(x*x*x, fragCoord, f3);
    
    fragColor = vec4(grid + f1 + f2 + f3, 1.0);
}
